Code

  Search.prototype.query = function(params, callback) {
    var self = this;

    if (params.q == null) {
      return callback(new Error(
        'Twitter:Search.query() requires q to be defined in the params'
      ));
    }

    self.api.get('/search/tweets.json', params, callback);
  };

  return Search;
})();
